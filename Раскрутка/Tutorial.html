$world_translate = new WorldTranslate(array(
    "basic_languages"=>"en",
    "notranslate"=>"",
    "using_ip"=>"not_define",
    "ajax_requests"=>"",
    "detect_language"=>""
));
 
// basic_languages - This language on where written the original content. 
 
// notranslate - This html blocks that do not need to translate. Indicate need id, class or just element.
 
// Example
 
$world_translate = new WorldTranslate(array(
    "basic_languages"=>"en",
    "notranslate"=>"#one,.two,span",
    "using_ip"=>"not_define",
    "ajax_requests"=>"",
    "detect_language"=>""
));
 
$html='
    <div id="one">This is just an example one</div>
    <span>This is just an example two</span>
        <b>Only this will be translated.</b>
    <p class="two">This is just an example three</p>
';
 
$html = $world_translate->get_world_translate($html,"en_to_de","en");
 
echo $html; /* 
    <div id="one">This is just one string</div>
    <span>This is just two string</span><br />
        <b>Dieser String wird übersetzt werden.</b>
    <div class="two">This is just three string</div>
*/
 
/* using_ip - Allows you to automatically determine the user's language and show him the page in his language. Can contain one of three options.
    1. not_define - language is not determined
    2. identify_ip - define the language using the ip Users
    3. determine_browser - define the language using the user's settings browser 
*/
 
// ajax_requests - If you want something to pages with specific url is not translated, simply specify the type of some unique information from the url.
